func safe(k int, positions []int) bool {
	for i, p := range positions {
		if p == k || abs(p-k) == i+1 {
			return false
		}
	}
	return true
}
func sequence(a, b int) []int {
	if a > b {
		return []int{}
	}
	result := make([]int, b-a+1)
	for i := range result {
		result[i] = a + i
	}
	return result
}
